Judul: program menghitung luas dan keliling lingkaran
Deklarasi: var pi, keliling, luas: double, var radius: integer

implementasi:
pi <- 3.14;
write("Masukkan radius:");
read(radius);

keliling <- 2 * pi * radius;
luas <- pi * radius * radius;

write("Keliling lingkaran: " + keliling);
write("Luas lingkaran: " + luas);
